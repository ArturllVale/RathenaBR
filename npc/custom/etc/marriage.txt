function SF_Groom {

		mes "Parece que sua noiva ainda não chegou e se registrou.";
		close;

		SF_AcceptBride();
		break;

		mes "Estou esperando seu parceiro aceitar você para iniciar a cerimônia.";
		close;

		SF_RetrieveRingM();
		break;

		close;

function SF_RetrieveRingM {
	mes "["+@name$+"]";
	if ($@wed_ring && countitem($@wed_ring) < 1) {
		mes "O que aconteceu com o seu "+getitemname($@wed_ring)+"? Você não o perdeu ... não é? Precisamos continuar com a cerimônia!";
		close;
	}
	if ($wed_bride_sex)
		set @item, 2634;	//Aliança do noivo
	else
		set @item, 2635;	//Aliança de noiva
	if (getnameditem(@item,$wed_groom$) == 0) {
		mes "Parece que você não tem espaço suficiente para carregar o anel... libere algum espaço e volte para recuperar o anel do seu parceiro.";
		close;
	}
	mes "Aqui está a aliança de casamento para sua noiva.";
	if ($@wed_ring) delitem $@wed_ring,1;
	set $wed_groom_progress,4;
	
	if ($wed_bride_progress == 4)
		SF_StartCeremony();
	else {
		mes "Assim que sua noiva recuperar o anel, a cerimônia começará.";
		close;
	}
}

function SF_StartCeremony {

	set $wed_groom_progress,5;

		set $@msg$,$wed_bride$;

		next;
		mes "["+@name$+"]";
		mes "Onde está "+$@msg$+"?? Eu não posso casar vocês dois se um está faltando...";
		close;
	}
	set $wed_bride_progress,6;
	set $wed_groom_progress,6;
	initnpctimer;
	close;
}

//Subfunção: Verifica se o noivo / noiva ainda está usando suas coisas.
function SF_equip_check {
	if (Sex == SEX_MALE && getequipid(EQI_ARMOR) != 7170) {
		mes "["+@name$+"]";
		mes "Criança, o que você fez com o seu "+getitemname(7170)+"?";
		emotion ET_THINK;
		return 0;
	}
	if (Sex == SEX_FEMALE && getequipid(EQI_ARMOR) != 2338) {
		mes "["+@name$+"]";
		mes "Criança, você deve usar um "+getitemname(2338)+" em todos os momentos durante a cerimônia...";
		emotion ET_THINK;
		return 0;
	}
	if (Sex == SEX_FEMALE && $@wed_veil && getequipid(EQI_HEAD_TOP) != 2206) {
		mes "["+@name$+"]";
		mes "Criança, você não pode tirar seu "+getitemname(2206)+" ainda....";
		emotion ET_THINK;
		return 0;
	}
	return 1;
}

		mes "Muito bem, você vai se registrar como o noivo?";
		next;
		if (select("^66cc66Sim^000000","^993333Mudei de ideia.^000000")==1)
			set @submenu, 1;
		else
			set @submenu, 3;
	} else { //Noiva

		mes "Muito bem, você vai se registrar como a Noiva?";
		next;
		if (select("^66cc66Sim^000000","^993333Mudei de ideia.^000000")==1)
			set @submenu, 2;
		else
			set @submenu, 3;
	}
	switch (@submenu) {
	case 1: //Noivo
		SF_TryRegister(0);
		set $wed_progress,1;
		mes "["+@name$+"]";
		mes "Agora você está registrado como noivo.";
		mes "Diga a sua noiva para se registrar o mais rápido possível.";

		mes "Diga ao seu noivo para se registrar o mais rápido possível.";

function SF_TryRegister {
	set @bride, getarg(0);
	set @type$, "noivo";
	if (@bride)
		set @type$, "noiva";
		
	mes "["+@name$+"]";
	mes "Antes de se registrar como "+@type$+", deixe-me verificar se você atende a todos os requisitos ...";
	next;
	if (Upper == 2) {
		mes "["+@name$+"]";
		mes "Nossa, você é muito jovem para pensar em casamento!";
		emotion ET_SURPRISE;
		close;
	}
	if (Sex == SEX_MALE)
		set @item, 7170;
	else
		set @item, 2338;
	
	if (getequipid(EQI_ARMOR) != @item) {
		mes "["+@name$+"]";
		mes "Você deveria estar usando um "+getitemname(@item)+" se você quer se casar.";
		close;
	}
	if (Sex == SEX_FEMALE && $@wed_veil && getequipid(EQI_HEAD_TOP) != 2206) {
		mes "["+@name$+"]";
		mes "Onde está seu "+getitemname(2206)+"? É um complemento necessário ao seu vestido.";
		close;
	}
	if ($@wed_ring && countitem($@wed_ring) < 1) {
		mes "["+@name$+"]";
		mes "Onde está o anel? Você precisa de um "+getitemname($@wed_ring)+" para a troca de anel, querida.";
		close;
	}
	if (@bride)
		set @cost, $@wed_bride_reg;
	else 
		set @cost, $@wed_groom_reg;

	if (Zeny < @cost) {
		mes "["+@name$+"]";
		mes "Sinto muito, mas você não tem o suficiente para pagar a taxa de registro.";
		mes "Volte depois de coletar "+@cost+"z."; 
		close;
	}
	set Zeny,Zeny-@cost;
	sc_start SC_Wedding,3600000,1; //Iniciar efeito de casamento (SC_WEDDING)
	if (@bride) {
		set $wed_bride_progress,1;
		set $wed_bride$,strcharinfo(0);
		set $wed_bride_sex, Sex;
		if ($@wedding_effect == 1) //ID da conta da loja para efeito.
			set $wedding_effect_id, getcharid(3);
	} else {
		set $wed_groom_progress,1;
		set $wed_groom$,strcharinfo(0);
		set $wed_groom_sex, Sex;
		if ($@wedding_effect == 2) //ID da conta da loja para efeito.
			set $wedding_effect_id, getcharid(3);
	}
}
